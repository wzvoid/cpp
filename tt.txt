#include <iostream>

using namespace std;

int main() {
    int n;
    cin >> n;
    int max = n / 9, min = INT32_MAX;
    for (int i = 0; i <= max; ++i) {
        int temp = n - 9 * i;
        if (temp % 4 == 0) {
            if (((temp / 4) + i) < min) {
                min = (temp / 4) + i;
            }
        }
    }
    if (min == INT32_MAX)
        cout << -1;
    else
        cout << min;
}
